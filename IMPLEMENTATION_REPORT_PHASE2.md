## 🔗 Интеграция в главное окно

### Пример использования в MainWindow:

```python
from ui.pages import (
    NotificationsSettingsPage, 
    SpecialDietsPage, 
    APIIntegrationPage
)

class MainWindow(QMainWindow):
    def __init__(self):
        super().__init__()
        self.init_ui()
        
    def init_ui(self):
        # Создать страницы
        self.notifications_page = self.create_notifications_page()
        self.diets_page = self.create_diets_page()
        self.api_page = self.create_api_page()
        
        # Добавить в navigation
        self.nav_buttons['notifications'].clicked.connect(
            lambda: self.stack_widget.setCurrentWidget(self.notifications_page)
        )
        
    def create_notifications_page(self):
        page = NotificationsSettingsPage()
        page.settings_saved.connect(self.save_notification_settings)
        return page
        
    def save_notification_settings(self, settings: dict):
        # Сохранить в БД или конфиг
        print(f"Saved: {settings}")
        
    def create_diets_page(self):
        page = SpecialDietsPage()
        page.diet_settings_changed.connect(self.update_diet_goals)
        return page
        
    def update_diet_goals(self, settings: dict):
        # Обновить цели пользователя
        if 'keto' in settings['active_diets']:
            self.user.carbs_goal = 50  # грамм
```
