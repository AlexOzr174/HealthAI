def create_demo_model(model_name, model_path):
    if model_name == "food_classifier.onnx":
        import numpy as np
        demo_data = b"ONNX_DEMO_MODEL_V1" + np.random.rand(100).tobytes()
        with open(model_path, 'wb') as f:
            f.write(demo_data)
